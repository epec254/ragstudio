Protos (internal only)

While I was writing these docs, I iterated on the rag_config.yml by writing ProtoBuf and then translating it to YAML. Eventually, the ProtoBuf got out of sync with the YAML, but since I don’t love YAML, and I think Proto is often more clear, so I asked ChatGPT to take the YAML configs in rag_config.yml and turn them into a proto.

Here it is in case they help clarify any confusion. This would NOT be in our user facing docs.

If anything that conflicts, use the YAML as the source of truth. we all know chatgpt hallucinates ;-)

syntax = "proto3";

package config;

// Main configuration file structure
message Config {
    GlobalConfig global_config = 1;
    EnvironmentConfig environment_config = 2;
    RAGComponentConfig rag_component_config = 3;
}

// Global configuration section
message GlobalConfig {
    string name = 1;
    string workspace_url = 2;
    string workspace_folder = 3;
    UnityCatalogConfig unity_catalog = 4;
    ArtifactLocationConfig artifact_location = 5;
    ArtifactLocationConfig output_data_location = 6;
    ArtifactLocationConfig logs_location = 7;
    repeated MetricConfig metrics = 8;
    string mlflow_experiment_name = 9;
    string vector_search_endpoint = 10;
}

// Unity Catalog configuration
message UnityCatalogConfig {
    string catalog = 1;
    string schema = 2;
}

// Artifact Location configuration
message ArtifactLocationConfig {
    string catalog = 1;
    string schema = 2;
}

// Metric configuration
message MetricConfig {
    string name = 1;
    string metric_name = 2;
    repeated LLMJudge llm_judges = 3;
    map<string, string> config = 4;
}

// LLM Judge configuration
message LLMJudge {
    string judge_name = 1;
    string endpoint_name = 2;
}

// Environment specific configuration
message EnvironmentConfig {
    map<string, EnvironmentOverride> end_users = 1;
    map<string, EnvironmentOverride> reviewers = 2;
    map<string, EnvironmentOverride> development = 3;
}

// Environment overrides
message EnvironmentOverride {
    string serving_endpoint = 1;
    string review_ui_name = 2;
    ArtifactLocationConfig output_data_location = 3;
    ArtifactLocationConfig logs_location = 4;
    repeated TableConfig tables = 5;
    repeated DataProcessorConfig data_processors = 6;
}

// Table configuration
message TableConfig {
    string request_log_table = 1;
    string assessment_log_table = 2;
    string online_evaluation_results_table = 3;
    string offline_evaluation_results_table = 4;
}

// Data Processor configuration
message DataProcessorConfig {
    string name = 1;
    string destination_table = 2;
    DestinationVectorIndexConfig destination_vector_index = 3;
    EmbeddingModelConfig embedding_model = 4;
    map<string, string> configurations = 5;
}

// Destination Vector Index configuration
message DestinationVectorIndexConfig {
    string endpoint_name = 1;
    string index_name = 2;
}

// Embedding Model configuration
message EmbeddingModelConfig {
    string endpoint_name = 1;
    string indexing_instruction = 2;
    string query_instruction = 3;
}

// RAG Component configuration
message RAGComponentConfig {
    repeated DataIngestorConfig data_ingestors = 1;
    repeated DataProcessorConfig data_processors = 2;
    repeated RetrieverConfig retrievers = 3;
    repeated ChainConfig chains = 4;
}

// Data Ingestor configuration
message DataIngestorConfig {
    string name = 1;
    string description = 2;
    string destination_uc_volume = 3;
}

// Retriever configuration
message RetrieverConfig {
    string name = 1;
    string description = 2;
    repeated string data_processors = 3;
    map<string, string> configurations = 4;
    string destination_model_name = 5;
}

// Chain configuration
message ChainConfig {
    string name = 1;
    string description = 2;
    repeated string retrievers = 3;
    repeated FoundationalModelConfig foundational_models = 4;
    string destination_model_name = 5;
    map<string, string> configurations = 6;
}

// Foundational Model configuration
message FoundationalModelConfig {
    string name = 1;
    string type = 2;
    string endpoint_name = 3;
    PromptTemplate prompt_template = 4;
}

// Prompt Template configuration
message PromptTemplate {
    repeated ChatMessage chat_messages = 1;
}

// Chat Message configuration
message ChatMessage {
    string role = 1;
    string content = 2;
}
